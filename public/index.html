          javascript
          3. Gate your operations by KYA score
        

        
      

      
async function checkBeforeOperation(agentAddress, operation) {
  const res = await fetch(
    `https://verun.network/api/score?address=${agentAddress}&operation=${operation}`
  );
  const kya = await res.json();

  if (!kya.permitted) {
    throw new Error(`KYA gate: score ${kya.score} insufficient for ${operation}`);
    // → 402 agent_not_verified
  }

  // proceed with operation
  return { agentScore: kya.score, kickback: kya.kickback };
}

// Usage
await checkBeforeOperation("GDRX...", "transfer"); // requires 500+

    

  







  

    
REST API

    
Endpoints

    

      

        

          GET
          /api/score
        

        
KYA score lookup. Returns score, tier, permissions, kickback rate, genesis identity, Stellar anchor.

        

          ?address=G...|0x...
          ?operation=read|mint|transfer|order
          no auth
        

      

      

        

          POST
          /api/register
        

        
Register as Genesis Agent. Runs Sybil check, anchors on Stellar Testnet, returns signed certificate.

        

          body: addressType, address, inviteCode?
          no auth
        

      

      

        

          GET
          /api/verify
        

        
Verify a Genesis claim by claim_id. Returns agent info + Stellar anchor + certificate status.

        

          ?claim_id=vga-0001
          no auth
        

      

      

        

          GET
          /api/stats
        

        
Live Genesis registration stats. Spots claimed, remaining, breakdown by type and tier.

        

          no params
          no auth
